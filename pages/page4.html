Practical Example

Example Code (Continue)

                        
                            const gab = [
                                'gab',
                                'lopez',
                                2003,
                                'student'
                            ];
    
                            for(let i = 0; i < gab.length; i++) {
                                if(typeof gab[i] !== 'string') continue; // Skip all data types that is not a string
                                console.log(gab[i], typeof gab[i]);
                            }

Example Code (Break)

                        
                            const gab = [
                                'gab',
                                'lopez',
                                2003,
                                'student'
                            ];
    
                            for(let i = 0; i < gab.length; i++) {
                                if(typeof gab[i] !== 'number') break; // First number data types is reached the for loops will break/stopped the iteration
                                console.log(gab[i], typeof gab[i]);
                            }
