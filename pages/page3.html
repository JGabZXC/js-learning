For Loop

a control flow statement that allows you to execute a block of code repeatedly

Syntax

                    
                        // for loop keeps running while condition is TRUE
                        for(initial value; logical condition; increment/decrement) {
                            // code block to be executed
                        }

Example Code

                    
                        for(let i = 1; i <= 10; i++) {
                            console.log(`Lifting weights repetition ${i} 🏋️‍♂️`);
                        }

Looping Arrays

Example Code

                        
                            const gab = [
                                'gab',
                                'lopez',
                                2003,
                                'student'
                            ];

                            for(let i = 0; i < gab.length; i++) {
                                console.log(gab[i]);
                            }

Creating an array

Example Code

                            
                                const types = [];
                                const gab = [
                                    'gab',
                                    'lopez',
                                    2003,
                                    'student'
                                ];
                                for(let i = 0; i < gab.length; i++) {
                                    // Reading from gab array
                                    console.log(gab[i]);
                                    // Filling types array
                                    types[i] = typeof gab[i];
                                }

Example Code

                            
                                const types = [];
    
                                const gab = [
                                    'gab',

                                    2003,

                                ];
    
                                for(let i = 0; i < gab.length; i++) {

Practical Example

Example Code

                            
                                const years = [1991, 2007, 1969, 2020];
                                const ages = [];

                                for(let i = 0; i < years.length; i++) {
                                    ages.push(2037 - years[i]);
                                }

                                console.log(ages); // Log the ages array

Looping Backwards and Loops in Loops

Example Code

                    
                        const gab = [
                            'gab',
                            'lopez',
                            2003,
                            'student',
                            ['Friend1','Friend2','Friend3']
                        ];

                        for(let i = gab.length - 1; i >= 0; i-- ) {
                            console.log(i, gab[i]);
                        }

Loops in Loops

                    
                        const gab = [
                            'gab',
                            'lopez',
                            2003,
                            'student',
                            ['Friend1','Friend2','Friend3']
                        ];

                        for(let i = gab.length - 1; i >= 0; i-- ) {
                            console.log(i, gab[i]);
                        }

                        for(let exercise = 1; exercise < 4; exercise++) {
                            console.log(`--------- Starting  exercise ${exercise}`);

                            for(let rep = 1; rep < 6; rep++) {
                                console.log(`Lifting weight repetition ${rep} 🏋️‍♂️`);
                            }
                        }
